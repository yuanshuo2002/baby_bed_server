> **目标**：让硬件系统与 Baby Bed Server 小程序后端对接
>
> **基准**：Baby Bed Server 当前版本 (2026-06-10)

## 一、现状分析

### 1.1 硬件端当前状态

| 模块 | 当前实现 | 问题 |
|------|---------|------|
| 设备身份 | `device_identity.txt` 写死 SN | 无法动态配置 |
| 雷达数据上传 | 硬编码 `device_id=22, baby_id=92` | 与小程序解耦后无法工作 |
| API 地址 | 硬编码 `223.247.96.246:8123` | 无法切换环境 |
| 设备注册 | **缺失** | 硬件上电后需先注册才能被小程序绑定 |
| 模式切换 | **缺失** | 小程序切换睡床/游戏床模式时硬件无法响应 |
| 状态上报 | **缺失** | 小程序无法查询设备在线状态 |
| 心跳机制 | **缺失** | 无法保持设备在线状态 |

### 2.1 后端 API需求（需要后端开发实现）

后端需新增以下 API（硬件端调用，无需用户认证）：

#### 2.1.1 心跳接口 `POST /hardware/heartbeat`

**功能**：硬件端定期发送心跳，后端返回设备绑定状态

**请求**：
```json
{
    "device_sn": "BB20240001",
    "work_mode": "sleep_bed",
    "online_status": 1,
    "cpu_percent": 25.5,
    "memory_percent": 45.2
}
```

**响应**：
```json
{
    "code": 0,
    "message": "success",
    "data": {
        "device_sn": "BB20240001",
        "is_bound": true,
        "baby_id": 92,
        "family_id": 10,
        "work_mode": "sleep_bed",
        "online_status": 1,
        "last_bind_at": "2026-06-10T15:30:00"
    }
}
```

**实现要点**：
- 更新设备 `last_online_at` 时间戳
- 设置设备 `online_status = 1`
- 查询设备绑定的 baby_id 并返回
- 如果设备未绑定，返回 `is_bound: false, baby_id: null`

#### 2.1.2 后端硬件路由修改 `app/api/v1/hardware.py`

新增文件或修改现有文件：

```python
"""
硬件交互路由 - 心跳与设备状态
"""
from fastapi import APIRouter, Body
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update

from app.api.deps import get_db_session
from app.core.response import success
from app.models.device import Device
from app.models.baby import Baby
from app.schemas.base import ApiResponse

router = APIRouter(prefix="/hardware", tags=["硬件交互"])


@router.post("/heartbeat", response_model=ApiResponse, summary="设备心跳 [硬件接口]")
async def device_heartbeat(
    device_sn: str = Body(..., description="设备序列号"),
    work_mode: str | None = Body(None, description="当前工作模式"),
    online_status: int | None = Body(1, description="在线状态"),
    cpu_percent: float | None = Body(None, description="CPU使用率"),
    memory_percent: float | None = Body(None, description="内存使用率"),
    db: AsyncSession = Depends(get_db_session),
):
    """
    设备心跳接口（硬件端调用，无需用户认证）
    同时返回设备绑定状态 baby_id
    """
    from datetime import datetime

    # 查询设备
    result = await db.execute(select(Device).where(Device.device_sn == device_sn))
    device = result.scalar_one_or_none()

    if not device:
        return success(data={
            "device_sn": device_sn,
            "is_bound": False,
            "baby_id": None,
            "family_id": None,
            "work_mode": work_mode,
            "online_status": 0,
        })

    # 更新设备状态
    device.last_online_at = datetime.now()
    device.online_status = online_status or 1
    if work_mode:
        device.work_mode = work_mode

    # 查询绑定状态
    baby_id = device.baby_id
    family_id = None
    is_bound = baby_id is not None

    if baby_id:
        result = await db.execute(select(Baby).where(Baby.id == baby_id))
        baby = result.scalar_one_or_none()
        if baby:
            family_id = baby.family_id

    return success(data={
        "device_sn": device_sn,
        "is_bound": is_bound,
        "baby_id": baby_id,
        "family_id": family_id,
        "work_mode": device.work_mode,
        "online_status": device.online_status,
        "last_bind_at": device.updated_at.isoformat() if device.updated_at else None,
    })
```

### 2.2 硬件端心跳实现

#### 2.2.1 增强版心跳循环

**文件**: `app/device_lifecycle.py` (修改)

```python
"""设备生命周期管理：注册、绑定、心跳 baby_id 获取"""
import threading
import time
import logging
from typing import Optional, Callable

from adapters.cloud_api.device_client import DeviceAPIClient
from config.device_identity import get_device_identity

logger = logging.getLogger(__name__)


class DeviceLifecycle:
    def __init__(
        self,
        on_mode_changed: Callable[[str], None] = None,
        on_bind_changed: Callable[[bool, int], None] = None,
    ):
        self.api_client = DeviceAPIClient()
        self.identity = get_device_identity()
        self._heartbeat_thread: Optional[threading.Thread] = None
        self._stop_heartbeat = threading.Event()
        self._current_mode = "sleep_bed"
        self._is_bound = False
        self._bound_baby_id: Optional[int] = None
        self._bound_family_id: Optional[int] = None
        self._on_mode_changed = on_mode_changed
        self._on_bind_changed = on_bind_changed
        self._last_heartbeat_response: dict = {}

    def register_and_start(self) -> bool:
        """注册设备并启动心跳"""
        try:
            # 注册设备
            result = self.api_client.register_device(
                device_sn=self.identity.device_sn,
                device_name=self.identity.device_name,
                device_model=self.identity.device_model,
            )
            logger.info(f"设备注册成功: {result}")

            # 启动心跳（包含绑定状态查询）
            self._start_heartbeat()
            return True
        except Exception as e:
            logger.error(f"设备注册失败: {e}")
            return False

    def _start_heartbeat(self) -> None:
        self._stop_heartbeat.clear()
        self._heartbeat_thread = threading.Thread(
            target=self._heartbeat_loop,
            daemon=True,
            name="DeviceHeartbeat",
        )
        self._heartbeat_thread.start()
        logger.info("设备心跳线程已启动 (每60秒)")

    def _heartbeat_loop(self) -> None:
        """心跳循环：发送心跳并解析绑定状态"""
        interval = 60  # 每60秒

        while not self._stop_heartbeat.wait(interval):
            try:
                # 发送心跳
                payload = {
                    "work_mode": self._current_mode,
                    "online_status": 1,
                }
                result = self.api_client.heartbeat(self.identity.device_sn, payload)

                # 解析返回的绑定状态
                self._handle_heartbeat_response(result)

            except Exception as e:
                logger.warning(f"心跳发送失败: {e}")

    def _handle_heartbeat_response(self, result: dict) -> None:
        """处理心跳响应，检测绑定状态变化"""
        try:
            data = result.get("data", {})
            is_bound = data.get("is_bound", False)
            baby_id = data.get("baby_id")
            family_id = data.get("family_id")

            # 检测绑定状态变化
            if is_bound and baby_id:
                if baby_id != self._bound_baby_id:
                    # baby_id 发生变化
                    old_baby_id = self._bound_baby_id
                    self._bound_baby_id = baby_id
                    self._bound_family_id = family_id
                    self._is_bound = True

                    logger.info(f"设备绑定状态变更: {old_baby_id} -> {baby_id}")
                    logger.info(f"  family_id: {family_id}")

                    # 回调通知
                    if self._on_bind_changed:
                        self._on_bind_changed(True, baby_id)

                    # 更新雷达上传器的 baby_id
                    self._update_radar_baby_id(baby_id)

            elif not is_bound:
                if self._is_bound:
                    # 从已绑定变为未绑定
                    self._is_bound = False
                    old_baby_id = self._bound_baby_id
                    self._bound_baby_id = None
                    self._bound_family_id = None

                    logger.info(f"设备解绑: baby_id={old_baby_id}")

                    if self._on_bind_changed:
                        self._on_bind_changed(False, None)

                    # 清空雷达上传器的 baby_id
                    self._update_radar_baby_id(None)

                else:
                    logger.debug("设备未绑定，等待用户扫码...")

            # 更新当前模式
            mode = data.get("work_mode")
            if mode and mode != self._current_mode:
                old_mode = self._current_mode
                self._current_mode = mode
                logger.info(f"模式变更: {old_mode} -> {mode}")
                if self._on_mode_changed:
                    self._on_mode_changed(mode)

            self._last_heartbeat_response = data

        except Exception as e:
            logger.error(f"处理心跳响应失败: {e}")

    def _update_radar_baby_id(self, baby_id: int | None) -> None:
        """更新雷达数据上传器的 baby_id"""
        try:
            # 导入雷达上传器
            from adapters.radar_api.uploader import RadarSensorUploadWorker

            # 通过全局方式更新（需要雷达上传器支持 set_baby_id）
            if hasattr(self, '_radar_mapper'):
                self._radar_mapper.set_baby_id(baby_id)
                logger.info(f"雷达上传器 baby_id 已更新: {baby_id}")
        except Exception as e:
            logger.warning(f"更新雷达 baby_id 失败: {e}")

    @property
    def is_bound(self) -> bool:
        return self._is_bound

    @property
    def baby_id(self) -> Optional[int]:
        return self._bound_baby_id

    @property
    def family_id(self) -> Optional[int]:
        return self._bound_family_id

    @property
    def current_mode(self) -> str:
        return self._current_mode

    def stop(self) -> None:
        self._stop_heartbeat.set()
        if self._heartbeat_thread:
            self._heartbeat_thread.join(timeout=2)
        logger.info("设备生命周期管理已停止")
```

### 2.3 设备状态持久化

**文件**: `app/device_state.py` (新建)

```python
"""设备状态持久化，保存 baby_id 等绑定信息到本地文件"""
import json
import threading
from pathlib import Path
from typing import Optional
import logging

logger = logging.getLogger(__name__)


class DeviceState:
    """设备状态单例，持久化存储绑定信息"""
    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        with cls._lock:
            if cls._instance is None:
                cls._instance = super().__new__(cls)
                cls._instance._initialized = False
            return cls._instance

    def __init__(self):
        if self._initialized:
            return
        self._state_file = Path(__file__).parent.parent / "device_state.json"
        self._lock = threading.Lock()
        self._state = {
            "is_bound": False,
            "baby_id": None,
            "family_id": None,
            "current_mode": "sleep_bed",
            "last_heartbeat_at": None,
        }
        self._load()
        self._initialized = True
        logger.info(f"DeviceState 已初始化，状态: {self._state}")

    def _load(self) -> None:
        """从文件加载状态"""
        if self._state_file.exists():
            try:
                with open(self._state_file, encoding="utf-8") as f:
                    self._state = json.load(f)
                logger.info(f"从文件加载设备状态: {self._state}")
            except Exception as e:
                logger.warning(f"加载设备状态失败: {e}")

    def _save(self) -> None:
        """保存状态到文件"""
        try:
            with open(self._state_file, "w", encoding="utf-8") as f:
                json.dump(self._state, f, ensure_ascii=False, indent=2)
        except Exception as e:
            logger.error(f"保存设备状态失败: {e}")

    @property
    def is_bound(self) -> bool:
        return self._state.get("is_bound", False)

    @property
    def baby_id(self) -> Optional[int]:
        return self._state.get("baby_id")

    @property
    def family_id(self) -> Optional[int]:
        return self._state.get("family_id")

    def bind(self, baby_id: int, family_id: int = None) -> None:
        """设备绑定成功"""
        with self._lock:
            self._state["is_bound"] = True
            self._state["baby_id"] = baby_id
            if family_id:
                self._state["family_id"] = family_id
            self._save()
        logger.info(f"设备已绑定: baby_id={baby_id}, family_id={family_id}")

    def unbind(self) -> None:
        """设备解绑"""
        with self._lock:
            self._state["is_bound"] = False
            self._state["baby_id"] = None
            self._state["family_id"] = None
            self._save()
        logger.info("设备已解绑")

    @property
    def current_mode(self) -> str:
        return self._state.get("current_mode", "sleep_bed")

    def set_mode(self, mode: str) -> None:
        """设置当前模式"""
        with self._lock:
            self._state["current_mode"] = mode
            self._save()

    def update_heartbeat(self) -> None:
        """更新心跳时间"""
        from datetime import datetime
        with self._lock:
            self._state["last_heartbeat_at"] = datetime.now().isoformat()
            self._save()
```

```python
import os
import json

@dataclass
class ApiEndpoints:
    # 基础配置
    base_url: str = "http://127.0.0.1:8000"
    api_prefix: str = "/api/v1"

    # 设备管理
    device_register: str = "/device/register"
    device_bind: str = "/device/bind"
    device_heartbeat: str = "/hardware/heartbeat"
    device_status: str = "/device/status/{device_sn}"

    # 传感器
    sensor_upload: str = "/sensor/upload"

    # 硬件控制
    mode_switch: str = "/hardware/mode/switch"
    mode_current: str = "/hardware/mode/current"

    # 系统
    storage_status: str = "/system/storage-status"
    performance_upload: str = "/system/performance/upload"
    health_upload: str = "/system/health/upload"

    def full_url(self, path: str) -> str:
        return f"{self.base_url}{self.api_prefix}{path}"

@dataclass
class ApiConfig:
    endpoints: ApiEndpoints
    timeout_sec: float = 10.0
    retry_count: int = 3
    retry_delay_sec: float = 2.0

def load_api_config() -> ApiConfig:
    config_path = Path(__file__).parent.parent / "config.json"
    if not config_path.exists():
        return ApiConfig(endpoints=ApiEndpoints())

    with open(config_path, encoding="utf-8") as f:
        data = json.load(f)

    api_section = data.get("api", {})
    endpoints = ApiEndpoints(
        base_url=api_section.get("base_url", "http://127.0.0.1:8000"),
        api_prefix=api_section.get("api_prefix", "/api/v1"),
    )

    return ApiConfig(
        endpoints=endpoints,
        timeout_sec=float(api_section.get("timeout_sec", 10.0)),
        retry_count=int(api_section.get("retry_count", 3)),
        retry_delay_sec=float(api_section.get("retry_delay_sec", 2.0)),
    )
```

```python
"""与小程序后端的设备管理 API 交互"""
import requests
import time
from typing import Optional

from config.api_config import load_api_config

class DeviceAPIClient:
    def __init__(self):
        self.config = load_api_config()
        self.endpoints = self.config.endpoints

    def register_device(self, device_sn: str, device_name: str, device_model: str = None) -> dict:
        """注册设备到云端"""
        url = self.endpoints.full_url(self.endpoints.device_register)
        payload = {
            "device_sn": device_sn,
            "device_name": device_name,
            "device_model": device_model or "unknown",
        }

        resp = requests.post(url, json=payload, timeout=self.config.timeout_sec)
        resp.raise_for_status()
        return resp.json()

    def heartbeat(self, device_sn: str, status: dict = None) -> dict:
        """发送设备心跳"""
        url = self.endpoints.full_url(self.endpoints.device_heartbeat)
        payload = {"device_sn": device_sn}
        if status:
            payload.update(status)

        resp = requests.post(url, json=payload, timeout=self.config.timeout_sec)
        resp.raise_for_status()
        return resp.json()

    def get_current_mode(self, device_sn: str) -> dict:
        """获取设备当前工作模式"""
        url = self.endpoints.full_url(self.endpoints.mode_current)
        params = {"device_sn": device_sn}

        resp = requests.get(url, params=params, timeout=self.config.timeout_sec)
        resp.raise_for_status()
        return resp.json()

    def switch_mode(self, device_sn: str, mode: str, baby_id: int = None) -> dict:
        """切换设备工作模式"""
        url = self.endpoints.full_url(self.endpoints.mode_switch)
        payload = {"device_sn": device_sn, "mode": mode}
        if baby_id:
            payload["baby_id"] = baby_id

        resp = requests.post(url, json=payload, timeout=self.config.timeout_sec)
        resp.raise_for_status()
        return resp.json()

    def get_storage_status(self, device_sn: str) -> dict:
        """获取存储状态"""
        url = self.endpoints.full_url(self.endpoints.storage_status)
        params = {"device_sn": device_sn}

        resp = requests.get(url, params=params, timeout=self.config.timeout_sec)
        resp.raise_for_status()
        return resp.json()
```

```python
"""设备生命周期管理：注册、绑定、心跳"""
import threading
import time
import logging
from typing import Optional, Callable

from adapters.cloud_api.device_client import DeviceAPIClient
from config.device_identity import get_device_identity

logger = logging.getLogger(__name__)

class DeviceLifecycle:
    def __init__(
        self,
        on_mode_changed: Callable[[str], None] = None,
        on_bind_changed: Callable[[bool, int], None] = None,
    ):
        self.api_client = DeviceAPIClient()
        self.identity = get_device_identity()
        self._heartbeat_thread: Optional[threading.Thread] = None
        self._stop_heartbeat = threading.Event()
        self._mode_check_thread: Optional[threading.Thread] = None
        self._stop_mode_check = threading.Event()
        self._current_mode = "sleep_bed"  # 默认模式
        self._is_bound = False
        self._bound_baby_id: Optional[int] = None
        self._on_mode_changed = on_mode_changed
        self._on_bind_changed = on_bind_changed

    def register_and_start(self) -> bool:
        """注册设备并启动心跳"""
        try:
            # 注册设备
            result = self.api_client.register_device(
                device_sn=self.identity.device_sn,
                device_name=self.identity.device_name,
                device_model=self.identity.device_model,
            )
            logger.info(f"设备注册成功: {result}")

            # 启动心跳
            self._start_heartbeat()
            # 启动模式检查
            self._start_mode_check()
            return True
        except Exception as e:
            logger.error(f"设备注册失败: {e}")
            return False

    def _start_heartbeat(self) -> None:
        self._stop_heartbeat.clear()
        self._heartbeat_thread = threading.Thread(
            target=self._heartbeat_loop,
            daemon=True,
            name="DeviceHeartbeat",
        )
        self._heartbeat_thread.start()
        logger.info("设备心跳线程已启动")

    def _heartbeat_loop(self) -> None:
        interval =60  # 每60秒
        while not self._stop_heartbeat.wait(interval):
            try:
                status = {
                    "work_mode": self._current_mode,
                    "online_status": 1,
                }
                self.api_client.heartbeat(self.identity.device_sn, status)
                logger.debug("心跳发送成功")
            except Exception as e:
                logger.warning(f"心跳发送失败: {e}")

    def _start_mode_check(self) -> None:
        self._stop_mode_check.clear()
        self._mode_check_thread = threading.Thread(
            target=self._mode_check_loop,
            daemon=True,
            name="ModeCheck",
        )
        self._mode_check_thread.start()
        logger.info("模式检查线程已启动")

    def _mode_check_loop(self) -> None:
        interval = 10  # 每10秒检查一次
        while not self._stop_mode_check.wait(interval):
            try:
                result = self.api_client.get_current_mode(self.identity.device_sn)
                new_mode = result.get("data", {}).get("mode", "sleep_bed")
                if new_mode != self._current_mode:
                    old_mode = self._current_mode
                    self._current_mode = new_mode
                    logger.info(f"模式变更: {old_mode} -> {new_mode}")
                    if self._on_mode_changed:
                        self._on_mode_changed(new_mode)
            except Exception as e:
                logger.debug(f"模式检查失败: {e}")

    def switch_mode(self, mode: str) -> bool:
        """切换设备模式"""
        try:
            self.api_client.switch_mode(
                device_sn=self.identity.device_sn,
                mode=mode,
                baby_id=self._bound_baby_id,
            )
            self._current_mode = mode
            logger.info(f"模式切换成功: {mode}")
            return True
        except Exception as e:
            logger.error(f"模式切换失败: {e}")
            return False

    def stop(self) -> None:
        self._stop_heartbeat.set()
        self._stop_mode_check.set()
        if self._heartbeat_thread:
            self._heartbeat_thread.join(timeout=2)
        if self._mode_check_thread:
            self._mode_check_thread.join(timeout=2)
        logger.info("设备生命周期管理已停止")
```

```python
"""设备状态存储，保存 baby_id 等绑定信息"""
import json
import threading
from pathlib import Path
from typing import Optional

class DeviceState:
    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        with cls._lock:
            if cls._instance is None:
                cls._instance = super().__new__(cls)
                cls._instance._initialized = False
            return cls._instance

    def __init__(self):
        if self._initialized:
            return
        self._state_file = Path(__file__).parent.parent / "device_state.json"
        self._lock = threading.Lock()
        self._state = {
            "is_bound": False,
            "baby_id": None,
            "family_id": None,
            "current_mode": "sleep_bed",
        }
        self._load()
        self._initialized = True

    def _load(self) -> None:
        if self._state_file.exists():
            try:
                with open(self._state_file, encoding="utf-8") as f:
                    self._state = json.load(f)
            except Exception:
                pass

    def _save(self) -> None:
        try:
            with open(self._state_file, "w", encoding="utf-8") as f:
                json.dump(self._state, f, ensure_ascii=False, indent=2)
        except Exception:
            pass

    @property
    def is_bound(self) -> bool:
        return self._state.get("is_bound", False)

    @property
    def baby_id(self) -> Optional[int]:
        return self._state.get("baby_id")

    def bind(self, baby_id: int, family_id: int = None) -> None:
        with self._lock:
            self._state["is_bound"] = True
            self._state["baby_id"] = baby_id
            if family_id:
                self._state["family_id"] = family_id
            self._save()

    def unbind(self) -> None:
        with self._lock:
            self._state["is_bound"] = False
            self._state["baby_id"] = None
            self._state["family_id"] = None
            self._save()

    @property
    def current_mode(self) -> str:
        return self._state.get("current_mode", "sleep_bed")

    def set_mode(self, mode: str) -> None:
        with self._lock:
            self._state["current_mode"] = mode
            self._save()
```

## 十二、注意事项

### 7.1 网络异常处理
- 所有 API 调用需设置超时 (默认10秒)
- 失败时自动重试 (最多3次，间隔2秒)
- 重试失败后记录日志，不阻塞主流程

### 7.2 baby_id 获取时机
- 设备刚注册时 `baby_id = None`
- 用户在微信小程序绑定设备后，硬件端通过轮询获取
- 轮询间隔建议 5-10 秒

### 7.3 多设备支持
- 如果一个家庭有多个设备，每个设备独立运行 `DeviceLifecycle`
- 通过 `device_sn` 区分，存储在各自的 `device_state.json`

### 7.4 配置优先级
```
环境变量 > config.json > 代码默认值
```

例如：
```bash
export BABY_API_BASE_URL=https://api.production.com
export BABY_DEVICE_SN=BB20249999
python main.py
```
